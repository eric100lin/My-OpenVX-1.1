# set target name
set( TARGET_NAME_1 vxcpp )
set( TARGET_NAME_2 vxcpp_test )

include_directories( BEFORE
                     ${CMAKE_CURRENT_SOURCE_DIR}
					 ${CMAKE_SOURCE_DIR}/include
                     ${CMAKE_SOURCE_DIR}/kernels/c_model 
                     ${CMAKE_SOURCE_DIR}/kernels/debug 
                     ${CMAKE_SOURCE_DIR}/kernels/extras )
					 
if ((WIN32) OR (CYGWIN))
   add_definitions( ${OPENVX_BUILDING_EXPORT_DEF} )
endif ((WIN32) OR (CYGWIN))
					 
add_definitions( -DVX_EXPORTING )
					 
FIND_SOURCES( vx.cpp )

# add a target named ${TARGET_NAME}
add_library (${TARGET_NAME_1} SHARED ${SOURCE_FILES})

target_link_libraries( ${TARGET_NAME_1} openvx-debug-lib openvx-c_model-lib openvx-debug_k-lib  openvx-extras_k-lib openvx-helper openvx )

add_executable(${TARGET_NAME_2} vx.cpp)

target_link_libraries( ${TARGET_NAME_2} openvx-debug-lib openvx-helper vxcpp openvx )

install ( TARGETS ${TARGET_NAME_1} ${TARGET_NAME_2}
          RUNTIME DESTINATION bin
          ARCHIVE DESTINATION lib
          LIBRARY DESTINATION bin )
		  
set_target_properties( ${TARGET_NAME_1} PROPERTIES FOLDER ${SAMPLE_CPP_FOLDER} )
set_target_properties( ${TARGET_NAME_2} PROPERTIES FOLDER ${SAMPLE_CPP_FOLDER} )
